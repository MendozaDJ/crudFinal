-- PARAMETROS_SISTEMA
CREATE OR REPLACE TRIGGER TRG_BI_PARAMETROS_SISTEMA
BEFORE INSERT ON PARAMETROS_SISTEMA
FOR EACH ROW
BEGIN
  IF :NEW.id_parametro IS NULL THEN
    SELECT SEQ_PARAMETROS_SISTEMA.NEXTVAL INTO :NEW.id_parametro FROM dual;
  END IF;
END;
/
